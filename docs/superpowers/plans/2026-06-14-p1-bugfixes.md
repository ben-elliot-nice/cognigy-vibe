```python
def _build_hierarchy(chart: dict) -> str:
    nodes = {n["_id"]: n for n in chart.get("nodes", [])}
    # Support both real AU1 API fields (node/next/children) and legacy mock fields (nodeId/nextId/childIds)
    relations = {
        (r.get("node") or r.get("nodeId") or r.get("_id")): r
        for r in chart.get("relations", [])
    }

    def render(node_id: str, indent: int = 0, visited: frozenset = frozenset()) -> list[str]:
        if node_id in visited:
            return [f"{'  ' * indent}[CYCLE → {node_id}]"]
        visited = visited | {node_id}
        node = nodes.get(node_id, {})
        label = node.get("label") or node.get("type", node_id)
        ntype = node.get("type", "")
        prefix = "  " * indent
        lines = [f"{prefix}[{ntype}] {label} ({node_id})"]
        rel = relations.get(node_id, {})
        for child_id in (rel.get("children") or rel.get("childIds", [])):
            lines += render(child_id, indent + 1, visited)
        next_id = rel.get("next") or rel.get("nextId")
        if next_id:
            lines += render(next_id, indent, visited)
        return lines

    # Root: a node not referenced as next or child by any other relation
    all_referenced: set[str] = set()
    for r in chart.get("relations", []):
        nxt = r.get("next") or r.get("nextId")
        if nxt:
            all_referenced.add(nxt)
        for child in (r.get("children") or r.get("childIds") or []):
            all_referenced.add(child)
    roots = [nid for nid in relations if nid not in all_referenced]
    if not roots and relations:
        roots = [next(iter(relations))]  # fallback for cyclic graphs
    lines = []
    for root in roots:
        lines += render(root)
    return "\n".join(lines) if lines else "(empty chart)"
```
